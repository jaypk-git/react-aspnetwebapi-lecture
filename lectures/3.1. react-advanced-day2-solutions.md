## 1. 고급 Hooks 사용법 - 장바구니 기능 구현

```jsx
import React, { createContext, useContext, useReducer } from 'react';

// 장바구니 Context 생성
const CartContext = createContext();

// 초기 상태
const initialState = {
  items: [],
  total: 0
};

// Reducer 함수
function cartReducer(state, action) {
  switch (action.type) {
    case 'ADD_ITEM':
      return {
        ...state,
        items: [...state.items, action.payload],
        total: state.total + action.payload.price
      };
    case 'REMOVE_ITEM':
      return {
        ...state,
        items: state.items.filter(item => item.id !== action.payload.id),
        total: state.total - action.payload.price
      };
    default:
      return state;
  }
}

// CartProvider 컴포넌트
function CartProvider({ children }) {
  const [state, dispatch] = useReducer(cartReducer, initialState);

  return (
    <CartContext.Provider value={{ state, dispatch }}>
      {children}
    </CartContext.Provider>
  );
}

// 상품 목록 컴포넌트
function ProductList() {
  const { dispatch } = useContext(CartContext);
  const products = [
    { id: 1, name: 'Product 1', price: 10 },
    { id: 2, name: 'Product 2', price: 20 },
    { id: 3, name: 'Product 3', price: 30 },
  ];

  return (
    <div>
      <h2>Products</h2>
      {products.map(product => (
        <div key={product.id}>
          {product.name} - ${product.price}
          <button onClick={() => dispatch({ type: 'ADD_ITEM', payload: product })}>
            Add to Cart
          </button>
        </div>
      ))}
    </div>
  );
}

// 장바구니 컴포넌트
function Cart() {
  const { state, dispatch } = useContext(CartContext);

  return (
    <div>
      <h2>Cart</h2>
      {state.items.map(item => (
        <div key={item.id}>
          {item.name} - ${item.price}
          <button onClick={() => dispatch({ type: 'REMOVE_ITEM', payload: item })}>
            Remove
          </button>
        </div>
      ))}
      <h3>Total: ${state.total}</h3>
    </div>
  );
}

// App 컴포넌트
function App() {
  return (
    <CartProvider>
      <div>
        <h1>Shopping App</h1>
        <ProductList />
        <Cart />
      </div>
    </CartProvider>
  );
}

export default App;
```

## 2. 커스텀 Hooks - useWindowSize

```jsx
import React, { useState, useEffect } from 'react';

// useWindowSize 커스텀 Hook
function useWindowSize() {
  const [windowSize, setWindowSize] = useState({
    width: undefined,
    height: undefined,
  });

  useEffect(() => {
    function handleResize() {
      setWindowSize({
        width: window.innerWidth,
        height: window.innerHeight,
      });
    }

    window.addEventListener('resize', handleResize);
    handleResize(); // 초기 크기 설정

    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return windowSize;
}

// 반응형 컴포넌트
function ResponsiveComponent() {
  const { width } = useWindowSize();

  if (width < 768) {
    return <div>Mobile Layout</div>;
  } else if (width < 1024) {
    return <div>Tablet Layout</div>;
  } else {
    return <div>Desktop Layout</div>;
  }
}

function App() {
  return (
    <div>
      <h1>Responsive App</h1>
      <ResponsiveComponent />
    </div>
  );
}

export default App;
```

## 3. 성능 최적화 기법 - 대량 데이터 리스트 최적화

```jsx
import React, { useState, useMemo, useCallback } from 'react';

// 리스트 아이템 컴포넌트
const ListItem = React.memo(({ item, onItemClick }) => {
  console.log('Rendering ListItem', item.id);
  return (
    <li onClick={() => onItemClick(item.id)}>
      {item.id}: {item.title}
    </li>
  );
});

// 리스트 컴포넌트
function List({ items }) {
  const [selectedId, setSelectedId] = useState(null);

  const handleItemClick = useCallback((id) => {
    setSelectedId(id);
  }, []);

  const memoizedItems = useMemo(() => items, [items]);

  return (
    <ul>
      {memoizedItems.map(item => (
        <ListItem 
          key={item.id} 
          item={item} 
          onItemClick={handleItemClick}
        />
      ))}
      {selectedId && <p>Selected ID: {selectedId}</p>}
    </ul>
  );
}

// App 컴포넌트
function App() {
  const [count, setCount] = useState(0);
  
  const items = useMemo(() => 
    Array.from({ length: 1000 }, (_, i) => ({
      id: i,
      title: `Item ${i}`
    })),
    []
  );

  return (
    <div>
      <h1>Optimized List</h1>
      <button onClick={() => setCount(c => c + 1)}>
        Increment Counter: {count}
      </button>
      <List items={items} />
    </div>
  );
}

export default App;
```

## 4. 고급 패턴과 베스트 프랙티스 - Mouse Tracker HOC

```jsx
import React, { useState, useEffect } from 'react';

// Mouse Tracker HOC
function withMouseTracker(WrappedComponent) {
  return function(props) {
    const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });

    useEffect(() => {
      function handleMouseMove(event) {
        setMousePosition({ x: event.clientX, y: event.clientY });
      }

      window.addEventListener('mousemove', handleMouseMove);

      return () => {
        window.removeEventListener('mousemove', handleMouseMove);
      };
    }, []);

    return <WrappedComponent {...props} mousePosition={mousePosition} />;
  };
}

// 마우스 위치를 표시하는 컴포넌트
function MouseDisplay({ mousePosition }) {
  return (
    <div>
      <h2>Mouse Position:</h2>
      <p>X: {mousePosition.x}, Y: {mousePosition.y}</p>
    </div>
  );
}

// 특정 영역에 들어갔을 때 동작하는 컴포넌트
function HighlightArea({ mousePosition }) {
  const isInside = mousePosition.x > 100 && mousePosition.x < 300 &&
                   mousePosition.y > 100 && mousePosition.y < 300;

  return (
    <div
      style={{
        width: '200px',
        height: '200px',
        position: 'absolute',
        left: '100px',
        top: '100px',
        backgroundColor: isInside ? 'yellow' : 'lightgray'
      }}
    >
      {isInside && <p>Mouse is inside!</p>}
    </div>
  );
}

const MouseTrackingDisplay = withMouseTracker(MouseDisplay);
const MouseTrackingArea = withMouseTracker(HighlightArea);

function App() {
  return (
    <div>
      <h1>Mouse Tracker App</h1>
      <MouseTrackingDisplay />
      <MouseTrackingArea />
    </div>
  );
}

export default App;
```

## 5. 고급 Todo 앱

```jsx
import React, { createContext, useContext, useReducer, useState, useCallback } from 'react';

// Context 생성
const TodoContext = createContext();

// 초기 상태
const initialState = {
  todos: [],
  filter: 'all'
};

// Reducer 함수
function todoReducer(state, action) {
  switch (action.type) {
    case 'ADD_TODO':
      return { 
        ...state, 
        todos: [...state.todos, { id: Date.now(), text: action.payload, completed: false }]
      };
    case 'TOGGLE_TODO':
      return { 
        ...state, 
        todos: state.todos.map(todo => 
          todo.id === action.payload ? { ...todo, completed: !todo.completed } : todo
        )
      };
    case 'REMOVE_TODO':
      return { 
        ...state, 
        todos: state.todos.filter(todo => todo.id !== action.payload)
      };
    case 'SET_FILTER':
      return { ...state, filter: action.payload };
    default:
      return state;
  }
}

// 로컬 스토리지 커스텀 Hook
function useLocalStorage(key, initialValue) {
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.log(error);
      return initialValue;
    }
  });

  const setValue = value => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    } catch (error) {
      console.log(error);
    }
  };

  return [storedValue, setValue];
}

// TodoProvider 컴포넌트
function TodoProvider({ children }) {
  const [state, dispatch] = useReducer(todoReducer, initialState);
  const [persistedState, setPersistedState] = useLocalStorage('todoState', state);

  React.useEffect(() => {
    setPersistedState(state);
  }, [state, setPersistedState]);

  return (
    <TodoContext.Provider value={{ state: persistedState, dispatch }}>
      {children}
    </TodoContext.Provider>
  );
}

// TodoInput 컴포넌트
const TodoInput = React.memo(function TodoInput() {
  const [input, setInput] = useState('');
  const { dispatch } = useContext(TodoContext);

  const handleSubmit = useCallback((e) => {
    e.preventDefault();
    if (input.trim()) {
      dispatch({ type: 'ADD_TODO', payload: input.trim() });
      setInput('');
    }
  }, [input, dispatch]);

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={input}
        onChange={(e) => setInput(e.target.value)}
        placeholder="Add a new todo"
      />
      <button type="submit">Add Todo</button>
    </form>
  );
});

// TodoItem 컴포넌트
const TodoItem = React.memo(function TodoItem({ todo }) {
  const { dispatch } = useContext(TodoContext);

  return (
    <li>
      <input
        type="checkbox"
        checked={todo.completed}
        onChange={() => dispatch({ type: 'TOGGLE_TODO', payload: todo.id })}
      />
      <span style={{ textDecoration: todo.completed ? 'line-through' : 'none' }}>
        {todo.text}
      </span>
      <button onClick={() => dispatch({ type: 'REMOVE_TODO', payload: todo.id })}>
        Remove
      </button>
    </li>
  );
});

// TodoList 컴포넌트
function TodoList() {
  const { state } = useContext(TodoContext);

  const filteredTodos = useMemo(() => {
    switch (state.filter) {
      case 'active':
        return state.todos.filter(todo => !todo.completed);
      case 'completed':
        return state.todos.filter(todo => todo.completed);
      default:
        return state.todos;
    }
  }, [state.todos, state.filter]);

  return (
    <ul>
      {filteredTodos.map(todo => (
        <TodoItem key={todo.id} todo={todo} />
      ))}
    </ul>
  );
}

// FilterButtons 컴포넌트
function FilterButtons() {
  const { state, dispatch } = useContext(TodoContext);

  return (
    <div>
      <button onClick={() => dispatch({ type: 'SET_FILTER', payload: 'all' })}>
        All
      </button>
      <button onClick={() => dispatch({ type: 'SET_FILTER', payload: 'active' })}>
        Active
      </button>
      <button onClick={() => dispatch({ type: 'SET_FILTER', payload: 'completed' })}>
        Completed
      </button>
      <p>Current filter: {state.filter}</p>
    </div>
  );
}

// App 컴포넌트
function App() {
  return (
    <TodoProvider>
      <div>
        <h1>Advanced Todo App</h1>
        <TodoInput />
        <FilterButtons />
        <TodoList />
      </div>
    </TodoProvider>
  );
}

export default App;
```

이 고급 Todo 앱은 다음과 같은 기능을 포함하고 있습니다:

1. Context API를 사용한 전역 상태 관리
2. useReducer를 활용한 복잡한 상태 로직 처리
3. 커스텀 Hook (useLocalStorage)을 사용한 로컬 스토리지 연동
4. React.memo, useMemo, useCallback을 활용한 성능 최적화
5. 컴포넌트 분리를 통한 재사용 가능한 구조
